def bmi():
    name = input("Enter your name: ")
    height = float(input("Enter height in inches: "))*0.0254
    weight = float(input("Enter weight in kgs: "))
    bmi = weight/(height**2)
    print("Body mass index for {0} is: {1}".format(name.lower(),bmi))
